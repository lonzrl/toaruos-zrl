#include <sys/stat.h>
#include <kuroko/vm.h>

KRK_Function(S_ISBLK) {
	int mode;
	if (!krk_parseArgs("i",(const char*[]){"mode"},&mode)) return NONE_VAL();
	return INTEGER_VAL(S_ISBLK(mode));
}
