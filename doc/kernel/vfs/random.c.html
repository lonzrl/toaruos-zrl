#include <stdint.h>
#include <kernel/vfs.h>
#include <kernel/string.h>
#include <kernel/time.h>

static uint32_t x = 123456789;
static uint32_t y = 362436069;
static uint32_t z = 521288629;
static uint32_t w = 88675123;

uint32_t rand(void) {
	uint32_t t;

	t = x ^ (x << 11);
	x = y; y = z; z = w;
	return w = w ^ (w >> 19) ^ t ^ (t >> 8);
}

static ssize_t read_random(fs_node_t *node, off_t offset, size_t size, uint8_t *buffer) {
	size_t remaining = size;
	size_t written = 0;
	while (remaining) {
		uint32_t r = rand();
		size_t to_write = remaining > 4 ? 4 : remaining;
		memcpy(&buffer[written], &r, to_write);
		remaining -= to_write;
		written += to_write;
	}
	return size;
}

static fs_node_t * random_device_create(void) {
	fs_node_t * fnode = malloc(sizeof(fs_node_t));
	memset(fnode, 0x00, sizeof(fs_node_t));
	fnode->inode = 0;
	strcpy(fnode->name, "random");
	fnode->uid = 0;
	fnode->gid = 0;
	fnode->mask = 0444;
	fnode->length  = 1024;
	fnode->flags   = FS_CHARDEVICE;
	fnode->read    = read_random;
	return fnode;
}

void random_initialize(void) {
	vfs_mount("/dev/random", random_device_create(), "random", "");
	vfs_mount("/dev/urandom", random_device_create(), "random", "");
	uint32_t seed = now();
	x = 123456789  ^ (seed << 16) ^ (seed >> 16);
}
