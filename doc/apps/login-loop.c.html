char * get_arg(FILE * f) {
	static char buf[32];
	int n = fgetc(f);
	if (n == '{') {
		int count = 0;
		do {
			int x = fgetc(f);
			if (x == '}') break;
			if (x < 0) break;
			buf[count++] = x;
			buf[count] = '\0';
		} while (count < 31);
		if (count) return buf;
	} else {
		ungetc(n, f);
	}
	return NULL;
}

char * get_ipv4_address(char * arg) {
	if (arg) {
		char if_path[1024];
		snprintf(if_path, 300, "/dev/net/%s", arg);
		int netdev = open(if_path, O_RDWR);
		if (netdev >= 0) {
			uint32_t ip_addr = 0;
			if (!ioctl(netdev, SIOCGIFADDR, &ip_addr)) {
				return inet_ntoa((struct in_addr){ip_addr});
			}
		}
	} else {
		/* Read /dev/net for interfaces */
		DIR * d = opendir("/dev/net");
		if (d) {
			struct dirent * ent;
			while ((ent = readdir(d))) {
				if (ent->d_name[0] == '.') continue;
				if (!strcmp(ent->d_name, "lo")) continue; /* Ignore loopback */
				closedir(d);
				return get_ipv4_address(ent->d_name);
			}
			closedir(d);
		}
	}

	return "127.0.0.1";
}

int main(int argc, char * argv[]) {
	while (1) {
		print_issue();
		pid_t f = fork();
		if (!f) {
			char * args[] = {
				"login",
				NULL
			};
			execvp(args[0], args);
			exit(2); /* Ensures login-loop halts */
			
		} else {
			int result, status;
			do {
				result = waitpid(f, &status, 0);
			} while (result < 0);
			if (WEXITSTATUS(status) == 2) break;
		}
	}

	return 1;
}
