#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>

int main(int argc, char ** argv) {
	if (argc < 2) {
		fprintf(stderr, "%s: missing operand\n", argv[0]);
		return 1;
	}

	char * arg = strdup(argv[1]);

	float time = atof(arg);
	useconds_t usecs = (useconds_t)(time * 1000000.0);

	if (usleep(usecs) < 0) {
		perror(argv[0]);
		return 1;
	}

	return 0;
}
