static void print_page_name(const char * page, const char * i) {
	int written = fprintf(stdout, "%s (%s)", page, i);
	while (written < 20) {
		fprintf(stdout, " ");
		written++;
	}
	fprintf(stdout, " - ");

static int try_filename(char * filename, char * page, char * i) {
	struct stat st;
	if (!stat(filename, &st)) {
		if (where_is) {
			puts(filename);
			return 1;
		}
		int is_gz = strlen(filename) > 3 && !strcmp(filename + strlen(filename)-3,".gz");
		char * systemcmd;
		if (what_is) {
			print_page_name(page, i);
			asprintf(&systemcmd,
				!is_gz ?
				(ROFF_CMD " -S NAME -P -I1 '%s' | head -n1") :
				("gunzip -c '%s' | " ROFF_CMD " -S NAME -P -I1 -- - | head -n1"),
				filename);
		} else {
			asprintf(&systemcmd,
				!is_gz ?
				(ROFF_CMD " '%s' | " MORE_CMD) :
				("gunzip -c '%s' | " ROFF_CMD " -- - | " MORE_CMD),
				filename, page, i);
		}
		int result = system(systemcmd);
		if (result) exit(WEXITSTATUS(result));
		free(systemcmd);
		return 1;
	}
	return 0;
}

static char ** sections_from_string(char * str) {
	size_t count = 1;
	char * c = str;
	for (; *c; ++c) {
		if (*c == ':') count++;
	}
	char ** sects = calloc(count + 1, sizeof(char*));
	size_t i = 0;
	c = str;
	while (*c) {
		sects[i] = c;
		while (*c && *c != ':') c++;
		if (*c == ':') {
			*c = '\0';
			i++;
			c++;
		}
	}

	return sects;
}

static int is_section(char * arg) {
	if (*arg >= 'a' && *arg <= 'z' && !arg[1]) return 1;
	if (*arg >= '0' && *arg <= '9') {
		arg++;
		while (*arg) {
			if (*arg < 'a' || *arg > 'z') return 0;
			arg++;
		}
		return 1;
	}
	return 0;
}
