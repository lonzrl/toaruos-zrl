#include <unistd.h>
#include <stdlib.h>
#include <stdio.h>
#include <string.h>
#include <errno.h>

int main(int argc, char ** argv) {
	char buf[NAME_MAX];

	if (getlogin_r(buf, NAME_MAX) == -1) {
		fprintf(stderr, "%s: failed to determine login name: %s\n", argv[0], strerror(errno));
		return 1;
	}

	fprintf(stdout, "%s\n", buf);
	return 0;
}
