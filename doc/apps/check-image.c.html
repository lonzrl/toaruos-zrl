#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <toaru/graphics.h>

int main (int argc, char * argv[]) {
	int quiet = 0;
	int opt;
	while ((opt = getopt(argc, argv, "q")) != -1) {
		switch (opt) {
			case 'q':
				quiet = 1;
				break;
			case '?':
				return 1;
		}
	}

	int ret = 0;
	while (optind < argc) {
		sprite_t * image = calloc(1, sizeof(sprite_t));
		if (load_sprite(image, argv[optind])) {
			if (!quiet) {
				fprintf(stdout, "%s: invalid\n", argv[optind]);
			}
			free(image);
			ret |= 1;
			optind++;
			continue;
		}

		if (!quiet) {
			fprintf(stdout, "%s: %d x %d\n", argv[optind], image->width, image->height);
		}

		sprite_free(image);
		optind++;
	}

	return ret;
}
