#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>

#define DEFAULT_PATH "/bin:/usr/bin"

int main(int argc, char * argv[]) {

	int ret_val = 0;
	int i = 1;
	int print_all = 0;

	if (i < argc && !strcmp(argv[i],"-a")) {
		print_all = 1;
		i++;
	}

	if (i == argc) return 1;

	for (; i < argc; ++i) {
		if (strstr(argv[i], "/")) {
			if (!access(argv[i], X_OK)) printf("%s\n", argv[i]);
		} else {
			char * file = argv[i];
			char * path = getenv("PATH") ?: DEFAULT_PATH;
			char * xpath = strdup(path);
			char * p, * last;
			int found = 0;
			for ((p = strtok_r(xpath, ":", &last)); p; p = strtok_r(NULL, ":", &last)) {
				char * exe = NULL;
				asprintf(&exe, "%s/%s", p, file);
				if (access(exe, X_OK)) {
					free(exe);
					continue;
				}
				found = 1;
				printf("%s\n", exe);
				free(exe);
				if (print_all) continue;
				break;
			}
			free(xpath);
			if (!found) ret_val = 1;
		}
	}
	return ret_val;
}
